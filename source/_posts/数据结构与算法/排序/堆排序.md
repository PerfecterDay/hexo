public class HeapSort {

    /**
     *  将arr数组看成是完全二叉树，arr[0]不保存数据
     *  建立初始大根堆
     * */
    private void buildMaxheap(int[] arr){
        int len = arr.length-1;
        /**
         * 最后一个节点的父节点是 i/2，从这个节点开始根节点，
         * 将调整以arr[j]为根的子树调整为堆
        * */
        for (int j = len/2; j > 0; j--) {
            adjustDown(arr,j,len);
        }
    }

    /**
     * k 指向树的节点
     * */
    private void adjustDown(int[] arr,int k, int len){
        arr[0] = arr[k];
        /**
         * arr[k]的左孩子是arr[2*k]，右孩子是arr[2*k+1]
         * 沿着孩子方向往下找
         * */
        for (int i = 2*k; i <= len; i = 2*i) {
            /**
             * 比较左右孩子的大小，i指向其中较大者
             * */
            if (i < len && arr[i] < arr[i+1]){
                i++; //右孩子大，i指向右孩子
            }
            if (arr[0] >= arr[i]){
                break; //根节点大于两个孩子，不用调整
            }else {
                //调换根节点和子树中较大的节点,调换后，k指向新的子树根节点或叶子节点
                arr[k] = arr[i];
                k = i;
            }
        }
        arr[k] = arr[0]; // k指向的节点是root的存放位置
    }


    /**
     * 堆排序算法，首先交换堆顶和堆底元素，然后将剩下的元素调整为大根堆，然后再交换堆顶和堆底...
     * */

    private void sort(int[] arr){
        buildMaxheap(arr);
        for (int i = arr.length-1; i > 1 ; i--) {
            int tmp = arr[i];
            arr[i] =  arr[1];
            arr[1] = tmp;

            adjustDown(arr,1,i-1);
        }
    }

    public static void main(String[] args) {
        int[] arr = new int[100];
        for (int i = 1; i < 100; i++) {
            arr[i] = i;
        }
        for (int i : arr) {
            System.out.print(i+",");
        }
        System.out.println();
        HeapSort heapSort = new HeapSort();
        heapSort.sort(arr);
        for (int i : arr) {
            System.out.print(i+",");
        }
        System.out.println();
    }
}
